API Example

// Check token sentiment
const response = await fetch('https://api.sentinel.ai/v1/sentiment', {
  method: 'POST',
  headers: {
    'Authorization': 'Bearer YOUR_API_KEY',
    'Content-Type': 'application/json'
  },
  body: JSON.stringify({
    token: 'SOL',
    timeframe: '24h'
  })
});

const data = await response.json();
console.log(data.sentiment); // { score: 0.85, label: 'Very Positive' }
